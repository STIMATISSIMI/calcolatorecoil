# Valori di resistività (Ω mm^2/m) RESISTIVITY = { 'SS316L': 0.00075, 'Kanthal A1': 1.45, 'Ni80': 1.09, } # Valori di TCR (Temperature Coefficient of Resistance) TCR = { 'SS316L': 0.00088, 'Kanthal A1': 0.000002, 'Ni80': 0.0001, } def calculate_resistance(material, wire_diameter, num_wraps, coil_length): """ Calcola la resistenza della coil. :param material: Materiale del filo (SS316L, Kanthal A1, Ni80) :param wire_diameter: Diametro del filo (mm) :param num_wraps: Numero di spire :param coil_length: Lunghezza della coil (mm) :return: Resistenza della coil (Ω) """ if material not in RESISTIVITY: raise ValueError("Materiale non valido. Scegli tra: SS316L, Kanthal A1, Ni80.") resistivity = RESISTIVITY[material] wire_area = (wire_diameter / 2) ** 2 * 3.1416 # Area della sezione del filo in mm^2 resistance_per_mm = resistivity / wire_area total_resistance = resistance_per_mm * coil_length * num_wraps return total_resistance def main(): print("Calcolatore della resistenza della coil") material = input("Inserisci il materiale del filo (SS316L, Kanthal A1, Ni80): ") wire_diameter = float(input("Inserisci il diametro del filo (mm): ")) num_wraps = int(input("Inserisci il numero di spire: ")) coil_length = float(input("Inserisci la lunghezza della coil (mm): ")) try: resistance = calculate_resistance(material, wire_diameter, num_wraps, coil_length) print(f"La resistenza della coil è: {resistance:.2f} Ω") except ValueError as e: print(e) if __name__ == "__main__": main()

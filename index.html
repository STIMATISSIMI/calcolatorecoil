import math # Resistività (ohm per metro) dei vari materiali resistivity = { 'Kanthal': 1.45e-6, 'Nichrome': 1.10e-6, 'Stainless Steel': 0.88e-6, 'Titanium': 0.35e-6 } def calculate_resistance(material, wire_diameter, coil_diameter, num_wraps): # Convertire il diametro del filo da millimetri a metri wire_diameter_m = wire_diameter * 1e-3 coil_diameter_m = coil_diameter * 1e-3 # Calcolare la lunghezza del filo in metri coil_circumference = math.pi * coil_diameter_m wire_length = coil_circumference * num_wraps # Calcolare l'area della sezione trasversale del filo cross_section_area = math.pi * (wire_diameter_m / 2) ** 2 # Calcolare la resistenza material_resistivity = resistivity[material] resistance = (material_resistivity * wire_length) / cross_section_area return resistance # Esempio di utilizzo material = 'Kanthal' wire_diameter = 0.5 # in millimetri coil_diameter = 2.5 # in millimetri num_wraps = 5 resistance = calculate_resistance(material, wire_diameter, coil_diameter, num_wraps) print(f"La resistenza della coil è: {resistance:.2f} ohm")
